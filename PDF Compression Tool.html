code Code Example 

// JavaScript Example - Ultra Compression
const formData = new FormData();
formData.append('file', fileInput.files[0]);
formData.append('mode', 'ultra');
formData.append('quality', '90');
formData.append('dpi', '200');
formData.append('colorCompression', '85');
formData.append('textOptimization', '95');
formData.append('objectCompression', '80');
formData.append('metadataRemoval', '90');

fetch('https://api.pdfcompressor.com/v1/ultra-compress', {
  method: 'POST',
  headers: {
    'Authorization': 'Bearer YOUR_API_KEY'
  },
  body: formData
})
.then(response => response.json())
.then(data => {
  console.log('Ultra compression successful:', data);
})
.catch(error => {
  console.error('Compression failed:', error);
});
